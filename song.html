실시간 구역 예약 시스템

1 브라우저 초기 기본 세팅

  1. 팝업 해제: 창이 하나만 뜨고 멈춘다면 주소창 우측 상단 팝업 차단 아이콘(🚫)을 눌러 "항상 허용"해 주세요.
  2. 개발자 권한: 크롬 주소창에 chrome://extensions/ 입력 이동 후, 우측 상단 [개발자 모드] 스위치를 켭니다(ON).
  3. 앱 세부 설정: Tampermonkey 확장앱 [세부정보] 클릭 후 [파일 URL에 대한 액세스 허용]을 활성화해 주세요.

2 Tampermonkey 전용 스크립트 주입 코드

아래 소스코드를 복사하여 크롬 Tampermonkey 대시보드 ➔ [새 스크립트 추가] 메뉴에 그대로 덮어쓰기하여 저장하세요.

// ==UserScript==
// @name         캠핑장 매크로창 정밀 조준 폭파 시스템
// @namespace    http://tampermonkey.net/
// @version      1.9
// @description  크로스 오리진 상황에서 마지막 창의 Alert가 렌더링을 차단하여 발생하는 잠금 현상을 완벽히 파괴합니다.
// @author       User
// @match        *://*.pubcamping.kr/*
// @run-at       document-start
// @grant        none
// @unwrap
// ==/UserScript==

(function() {
    'use strict';

    function killAlertSystem() {
        try {
            if (window.__alert_killed__) return;

            Object.defineProperty(window, 'alert', {
                value: function(msg) {
                    console.log("🎯 [매크로 엔진] 캠핑장 경고창 차단 성공:", msg);
                    try {
                        window.close();
                    } catch(e) {
                        window.location.href = "about:blank";
                    }
                    return true;
                },
                writable: false,
                configurable: false
            });

            Object.defineProperty(window, 'confirm', {
                value: function(msg) {
                    console.log("🎯 [매크로 엔진] 캠핑장 선택창 차단 성공:", msg);
                    try { window.close(); } catch(e) {}
                    return true;
                },
                writable: false,
                configurable: false
            });

            window.__alert_killed__ = true;
        } catch (err) {
            console.error("후킹 실패:", err);
        }
    }

    killAlertSystem();
    killAlertSystem();
    window.addEventListener('scroll', killAlertSystem, { once: true });
    window.addEventListener('mousemove', killAlertSystem, { once: true });

    if (!window.opener) return;

    var isMacroWindow = window.name && window.name.indexOf('macro_') === 0;
    if (!isMacroWindow) {
        console.log("[안전 구역] 메인 본창 또는 로그인 세션용 창이므로 강제 폐기를 면제합니다.");
        return;
    }

    console.log("🚀 [저격 모드 가동] 대상 매크로 타깃 윈도우 명칭: " + window.name);

    window.addEventListener("message", function(event) {
        var data = event.data;
        if (data && data.type === "MACRO_RESIZE_COMMAND") {
            try {
                window.resizeTo(data.width, data.height);
                window.moveTo(data.left, data.top);
            } catch (err) {
                console.error("창 크기 자체 조정 실패:", err);
            }
        }
    });

    function checkTargetUrl() {
        var currentUrl = window.location.href;
        if (currentUrl.indexOf('/@song/index') !== -1) {
            window.close();
        }
    }

    checkTargetUrl();
    window.addEventListener('DOMContentLoaded', checkTargetUrl);
})();
